/*SCRIPT DATOS DE CURSADA*/
CREATE DATABASE FACULTAD2021
GO
USE FACULTAD2021
GO
CREATE TABLE CARRERAS(
ID_CARRERA VARCHAR(4) NOT NULL PRIMARY KEY,
NOMBRE VARCHAR(25) NOT NULL,
CREACION_MATERIA DATE NOT NULL CHECK( CREACION_MATERIA <= GETDATE()),
MAIL VARCHAR(25) NOT NULL,
NIVEL VARCHAR(8) NULL  CHECK (NIVEL = 'Diplomatura' OR NIVEL = 'Pregrado' OR NIVEL = 'Grado' OR NIVEL = 'Posgrado'),
)
GO
CREATE TABLE MATERIAS(
ID_MATERIA INT NOT NULL PRIMARY KEY IDENTITY(1,1),
CARRERA VARCHAR(4) NOT NULL FOREIGN KEY REFERENCES CARRERAS(ID_CARRERA),
NOMBRE VARCHAR(25) NOT NULL,
CARGA_HORARIA TINYINT NOT NULL
)
GO
CREATE TABLE ALUMNOS (
LEGAJO INT NOT NULL PRIMARY KEY IDENTITY(1000,1),
CARRERA VARCHAR(4) NOT NULL FOREIGN KEY REFERENCES CARRERAS (ID_CARRERA),
APELLIDO VARCHAR(25) NOT NULL,
NOMBRE VARCHAR(25) NOT NULL,
FEC_NAC DATE NOT NULL CHECK (YEAR(GETDATE() - YEAR(FEC_NAC) )>17),
MAIL VARCHAR(25) NOT NULL UNIQUE,
TEL VARCHAR(20) NULL,
)
